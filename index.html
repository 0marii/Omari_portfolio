// Event-driven Jira sync via n8n
@Injectable()
export class ProductsService {
  async create(dto: CreateProductDto) {
    const product = await this.repo.save(dto);

    const issue = await this.n8n.trigger({
      event: 'jira.create',
      payload: product,
    });

    return { product, jiraKey: issue.key };
  }
}
